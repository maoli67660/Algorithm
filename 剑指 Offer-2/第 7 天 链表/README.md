### 第 15 天 队列
##### 树的层次遍历
```python
def level_order(root):
    queue = [root]
    level_num = 1
    res = []
    while queue:
        level_count = 0
        level_ls = []
        for _ in range(level_num):
            top = queue.pop(0)
            level_ls.append(top.val)
            if top.left:
                queue.append(top.left)
                level_count+=1
            if top.right:
                queue.append(top.right)
                level_count+=1
        level_num = level_count
        res.append(level_ls)
    return res
```

```python
def dfs(left, right, path, n ,res):
    if left==n and left==right:
        res.add(path)
        return
        
    if left >= right and left<=n:
        dfs(left+1, right, path+'(', n, res)
        dfs(left, right+1, path+')', n, res)
```

```python
def minCost(costs):
    dp = [[float('inf')]*3 for i in range(len(costs))]
    dp[0][0], dp[0][1], dp[0][2] = costs[0][0], costs[0][1], costs[0][2]
    
    for i in range(1, len(costs)):
        for j in range(3):
            for k in range(3):
                if k!=j:
                    dp[i][j] = min(dp[i][j], dp[i-1][k]+costs[i][j])
    return min(dp[-1])
```
